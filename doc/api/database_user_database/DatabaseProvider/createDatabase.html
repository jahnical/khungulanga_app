createDatabase() async {
  Directory documentsDirectory = await getApplicationDocumentsDirectory();
  String path = join(documentsDirectory.path, "User.db");

  var database = await openDatabase(
    path,
    version: 1,
    onCreate: initDB,
    onUpgrade: onUpgrade,
  );
  return database;
}
